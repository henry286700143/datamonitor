VOD_TIME	SELECT ROUND(SUM(VODTIME)/3600,2) AS VODTIME, SUM(PLAYTIMES) AS PLAYTIMES FROM VOD_TIME WHERE TIME = TO_DATE('${time}','YYYY-MM-DD') AND AREACODE = 'NULL'
CHANNEL_MINUTE_RTG	SELECT ROUND(SUM(RTG)/60,2) AS RTG FROM CHANNEL_MINUTE_RTG WHERE TIME >= TO_DATE('${time} 00:00:00','YYYY/MM/DD HH24:MI:SS') AND TIME <= TO_DATE('${time} 23:59:59','YYYY/MM/DD HH24:MI:SS') AND AREACODE = 'NULL'
ALL_MINUTE_RTG	SELECT SUM(RTG) AS RTG FROM ALL_MINUTE_RTG WHERE TIME >= TO_DATE('${time} 00:00:00', 'YYYY-MM-DD HH24:MI:SS') AND TIME <= TO_DATE('${time} 23:59:59', 'YYYY-MM-DD HH24:MI:SS') AND AREACODE = 'NULL'
C3_ORDER_ORDERFILE	SELECT COUNT(USERID) AS USERID, SUM(PRICE) AS PRICE FROM C3_ORDER_ORDERFILE WHERE PURCHASETIME >= TO_DATE('${time} 00:00:00', 'YYYY-MM-DD HH24:MI:SS') AND PURCHASETIME <= TO_DATE('${time} 23:59:59', 'YYYY-MM-DD HH24:MI:SS')
CHANNEL_DAY	SELECT ROUND(SUM(LENGTH) / 3600, 2) AS LENGTH, SUM(PLAYTIMES) AS PLAYTIMES FROM CHANNEL_DAY WHERE TIME_DAY = TO_DATE('${time}', 'YYYY-MM-DD') AND AREACODE = 'NULL'
CHANNEL_PROGRAM_LIST	SELECT SUM(RTG) AS RTG, SUM(SHR) AS SHR, SUM(RTG000) AS RTG000 FROM CHANNEL_PROGRAM_LIST WHERE AIRDATE = TO_DATE('${time}', 'YYYY-MM-DD') AND AREACODE = 'NULL'
PORTALVIEW	SELECT SUM(USERS) AS USERS, SUM(PLAYTIMES) AS PLAYTIMES, ROUND(SUM(TIMES) / 3600, 2) AS TIMES FROM PORTALVIEW WHERE TIME = TO_DATE('${time}', 'YYYY-MM-DD') AND AREACODE = 'NULL'
