VOD_ALL_TIME_STAT	SELECT ROUND(SUM(VODTIME)/3600,2) AS VODTIME, SUM(PLAYTIMES) AS PLAYTIMES FROM VOD_ALL_TIME_STAT WHERE TIME = TO_DATE('${time}','YYYY-MM-DD') AND AREACODE = 'NULL'
CHANNEL_MINUTE_RTG	SELECT ROUND(SUM(RTG)/60,2) AS RTG FROM CHANNEL_MINUTE_RTG WHERE TIME >= TO_DATE('${time} 00:00:00','YYYY/MM/DD HH24:MI:SS') AND TIME <= TO_DATE('${time} 23:59:59','YYYY/MM/DD HH24:MI:SS') AND AREACODE = 'NULL'
C3_ORDER_ORDERFILE	SELECT COUNT(USERID) AS USERID, SUM(PRICE) AS PRICE FROM C3_ORDER_ORDERFILE WHERE PURCHASETIME >= TO_DATE('${time} 00:00:00', 'YYYY-MM-DD HH24:MI:SS') AND PURCHASETIME <= TO_DATE('${time} 23:59:59', 'YYYY-MM-DD HH24:MI:SS')
TVOD_TIME_STAT	SELECT ROUND(SUM(TVODTIME) / 3600, 2) AS TVODTIME, SUM(TIMES) AS TIMES, SUM(PLAYTIMES) AS PLAYTIMES FROM TVOD_TIME_STAT WHERE TIME = TO_DATE('${time}', 'YYYY-MM-DD') AND AREACODE = 'NULL'
VOD_ALL_TIME_COL_STAT	SELECT ROUND(SUM(VODTIME) / 3600, 2) AS VODTIME, SUM(TIMES) AS TIMES, SUM(PLAYTIMES) AS PLAYTIMES FROM VOD_ALL_TIME_COL_STAT WHERE TIME = TO_DATE('${time}', 'YYYY-MM-DD') AND AREACODE = 'NULL'
VOD_ALL_TIME_SUBJECT_STAT	SELECT ROUND(SUM(VODTIME) / 3600, 2) AS VODTIME, SUM(TIMES) AS TIMES, SUM(PLAYTIMES) AS PLAYTIMES FROM VOD_ALL_TIME_SUBJECT_STAT WHERE TIME = TO_DATE('${time}', 'YYYY-MM-DD') AND AREACODE = 'NULL'
VOD_CONTENT	SELECT COUNT(CONTENTID) AS CONTENTID FROM VOD_CONTENT WHERE UPLOADTIME >= TO_DATE('${time} 00:00:00', 'YYYY-MM-DD HH24:MI:SS') AND UPLOADTIME <= TO_DATE('${time} 23:59:59', 'YYYY-MM-DD HH24:MI:SS')
C3_USERINFO_UFILE_CUR	SELECT SUM(ADD_USERS) AS ADD_USERS, SUM(CANCEL_USERS) AS CANCEL_USERS, SUM(NETGROWTH_USERS) AS NETGROWTH_USERS, SUM(PAUSE_USERS) AS PAUSE_USERS, SUM(ADD_HD_USERS) AS ADD_HD_USERS, SUM(ADD_SD_USERS) AS ADD_SD_USERS, SUM(CANCEL_HD_USERS) AS CANCEL_HD_USERS, SUM(CANCEL_SD_USERS) AS CANCEL_SD_USERS FROM C3_USERINFO_UFILE_CUR WHERE TIME = TO_DATE('${time}', 'YYYY-MM-DD') 	
C3_USERINFO_CVFILE_STAT_CUR	SELECT SUM(SILENCE_USERS) AS SILENCE_USERS, SUM(ONLINE_USERS) AS ONLINE_USERS, SUM(ADD_USERS) AS ADD_USERS, SUM(VIEWING_USERS) AS VIEWING_USERS, SUM(ADD_HD_USERS) AS ADD_HD_USERS, SUM(ADD_SD_USERS) AS ADD_SD_USERS FROM C3_USERINFO_CVFILE_STAT_CUR WHERE TIME = TO_DATE('${time}', 'YYYY-MM-DD')
